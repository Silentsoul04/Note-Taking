```python
# 检查一个正整数N是否为素数，最简单的方法就是试除法，将该数N用小于等于根号N的所有素数去试除，若均无法整除，N则为素数。

# 素数判断方法
def is_prime(n):
    if n % 2 == 0:
        return False

    sqrt_n = int(math.floor(math.sqrt(n)))
    for i in range(3, sqrt_n + 1, 2):
        if n % i == 0:
            return False
    return True
```

3. **最大公约数**：给你两个正整数a和b， 输出它们的最大公约数。例如：a = 3, b = 5，则输出：1

```python
def gcd(a, b):
    for i in range(1, min(a, b) + 1):
        if (a%i == 0) and (b%i == 0):
            gcd = i
    return gcd
print(gcd(a, b))
```

```python
def fib(num):
    a, b = 0, 1
    for i in range(num):
        yield a
        a, b = b, a+b

def fib(max):
    n, a, b = 0, 0, 1
    while n < max:
        yield a
        a, b = b, a + b
        n = n + 1
    return 'Done'
```

12. **二分查找**：

* 匹配的首个元素的下标

```python
def binary_search_left(lst, value):
    lo = 0
    hi = len(lst)
    while lo < hi:
        mid = (lo+hi) // 2
        if value > lst[mid]:
            lo = mid + 1	# 首次匹配到value时，lo值就是首次匹配的下标
        else:
            hi = mid
    if lst[lo] == value:
        return lo
    return None

def binary_search_bisect(lst, x):
    # 标准库 bisect
    import bisect
    i = bisect.bisect_left(lst, x)
    if lst[i] == x:
        return i
    return None
```

* 匹配的最后一个元素下标

```python
def binary_search_right(lst, value):
    lo = 0
    hi = len(lst)
    while lo < hi:
        mid =(lo+hi) // 2
        if value < lst[mid]:
            hi = mid
        else:
            lo = mid + 1	# 匹配到value后，lo值仍不断增加继续匹配，lo值最后是value下标+1
    if lst[lo-1] == value:
        return lo-1
    return None

def binary_search_bisect(lst, x):
    # 标准库 bisect
    import bisect
    i = bisect.bisect_right(lst, x)
    if lst[i-1] == x:
        return i-1
    return None
```

13. **冒泡排序**：

```python
def bubble_sort(iterable):
    lst = list(iterable)
    lst_len = len(lst)
    for i in range(lst_len-1):				#n个元素，只需循环n-1次
        for j in range(lst_len-1, i, -1):
            if lst[j] < lst[j-1]:
                lst[j], lst[j-1] = lst[j-1], lst[j]
    return lst
```

14. **插入排序**：

```python
def insertion_sort(iterable):
    lst = list(iterable)
    lst_len = len(lst)
    if lst_len==1: return lst
    for i in range(1, lst_len):
        for j in range(i, 0, -1):
            if lst[j] < lst[j-1]:
                lst[j], lst[j-1] = lst[j-1], lst[j]
            else:
                break
    return lst
```

17. **格雷码**：给定一个整数n，请返回n位连续的格雷码，顺序为从0开始。

* 递归

```python
def gray_code(n):
    gray = []
    if n == 1:
        gray.append('0')
        gray.append('1')
        return gray
    gray_pre = gray_code(n-1)
    for i in range(len(gray_pre)):
        gray.append('0' + gray_pre[i])
    for i in range(len(gray_pre)-1, -1, -1):
        gray.append('1' + gray_pre[i])
    return gray
```

* 循环

```python
def gray_code(n):
    gray = ['0', '1']
    for i in range(2, n+1): # n=2时，进入计算
        nums = pow(2, i)    # 下一个格雷码的长度
        gray_next = ['' for i in range(nums)]
        for j in range(nums):
            if j < (nums/2):
                gray_next[j] = '0' + gray[j]
            else:
                gray_next[j] = '1' + gray[nums-j-1]
        gray = gray_next
    return gray
```
